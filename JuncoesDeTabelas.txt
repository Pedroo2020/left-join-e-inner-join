/*			exercicio 1			*/
SELECT P.*, T.NOME, T.ENDERECO, T.TELEFONE
FROM PACIENTE p 
LEFT JOIN TUTOR t ON T.ID_TUTOR = P.ID_TUTOR 


/*			exercicio 2			*/
SELECT P.*, T.*
FROM PACIENTE p 
INNER JOIN TUTOR t ON T.ID_TUTOR = P.ID_TUTOR 


/*			exercicio 3			*/
SELECT T.*, P.NOME, P.ESPECIE, P.RACA
FROM TUTOR t 
LEFT JOIN PACIENTE p ON P.ID_TUTOR = T.ID_TUTOR 


/*			exercicio 4			*/
SELECT T.*, P.*
FROM TUTOR t
INNER JOIN PACIENTE p ON P.ID_TUTOR = T.ID_TUTOR 


/*			exercicio 5			*/
SELECT M.*, R.DATA_EMISSAO, I.QUANTIDADE, I.DOSAGEM
FROM MEDICAMENTOS m
LEFT JOIN ITENSRECEITA i ON i.ID_MEDICAMENTO = m.ID_MEDICAMENTO 
LEFT JOIN RECEITA r ON r.ID_RECEITA = i.ID_RECEITA


/*			exercicio 6			*/
SELECT M.*, R.*, i.*
FROM MEDICAMENTOS m
INNER JOIN ITENSRECEITA i ON i.ID_MEDICAMENTO = m.ID_MEDICAMENTO 
LEFT JOIN RECEITA r ON r.ID_RECEITA = i.ID_RECEITA


/*			exercicio 7			*/
SELECT r.DATA_EMISSAO
	   , i.QUANTIDADE
	   , i.DOSAGEM
	   , p.NOME
	   , p.ESPECIE
 FROM RECEITA r
 LEFT JOIN AGENDAMENTO a  ON a.ID_AGENDAMENTO = r.ID_AGENDAMENTO
 LEFT JOIN PACIENTE p     ON p.ID_PACIENTE    = a.ID_PACIENTE 
 LEFT JOIN ITENSRECEITA i ON i.ID_RECEITA     = r.ID_RECEITA

/*			exercicio 8			*/
SELECT r.*
	   , i.*
	   , p.*
 FROM RECEITA r
 INNER JOIN AGENDAMENTO a  ON a.ID_AGENDAMENTO = r.ID_AGENDAMENTO
 INNER JOIN PACIENTE p     ON p.ID_PACIENTE    = a.ID_PACIENTE 
 INNER JOIN ITENSRECEITA i ON i.ID_RECEITA     = r.ID_RECEITA


/*			exercicio 9			*/
SELECT DISTINCT r.*, m.NOME, i.DOSAGEM 
FROM RECEITA r
LEFT JOIN ITENSRECEITA i ON i.ID_RECEITA = r.ID_RECEITA 
LEFT JOIN MEDICAMENTOS m ON m.ID_MEDICAMENTO = i.ID_MEDICAMENTO 


/*			exercicio 10			*/
SELECT DISTINCT r.*
	  , m.*
	  , i.QUANTIDADE 
	  , i.DOSAGEM 
FROM RECEITA r
INNER JOIN ITENSRECEITA i ON i.ID_RECEITA     = r.ID_RECEITA 
INNER JOIN MEDICAMENTOS m ON m.ID_MEDICAMENTO = i.ID_MEDICAMENTO
